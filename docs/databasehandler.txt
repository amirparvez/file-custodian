-   database-handler: is used to manage a database.
#   Config: { system, host, port, database, username, password, table_name, proper_delete, sequelize_instance, user_model }

#   setupModel (private) : Syncs File Model and relates with User model if required.
    returns true or false

#   createTable : Creates new table.
    returns true or false

#   connectToDatabase (private) : Connects to the database.
    returns true or false

#   getModel (private) : Gets file entry from the database.
    input : { path }
    returns file-model or null

#   newModel (private) : Creates file entry in the database.
    input : { name, ext, folder, size, isEncrypted, user_id, }
    returns file-model or null

#   deleteModel (private) : Deletes/Updates file entry from/in the database.
    input : { (id)/path }
    returns true or false

#   updateModel (private) : Updates file entry in the database.
    input : { (id)/path, (new_name)/isEncrypted }
    returns true or false
